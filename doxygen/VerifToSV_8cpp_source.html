38  // Printf's will be emitted to stdout (32'h8000_0001 in IEEE Std 1800-2012).
39  Value fdStdout = rewriter.create<hw::ConstantOp>(
40  op.getLoc(), APInt(32, 0x80000001, false));

44  if (!fstrOp)

56 void VerifToSVPass::runOnOperation() {
57  MLIRContext &context = getContext();
58  hw::HWModuleOp module = getOperation();
59 
60  ConversionTarget target(context);
61  RewritePatternSet patterns(&context);
62 
63  target.addIllegalOp<PrintOp>();
64  target.addLegalDialect<sv::SVDialect, hw::HWDialect>();
65  patterns.add<PrintOpConversionPattern>(&context);
66 
67  if (failed(applyPartialConversion(module, target, std::move(patterns))))
68  signalPassFailure();
