60 /// Helper to convert a Value while building up a ClassOp. If the Value is
61 /// already in the IRMapping, it's already been converted. If the Value is
62 /// defined by an op, one of two things happens. If the op is an InstanceOp, the
63 /// instance is converted to an ObjectOp and the output field accessed. If the
64 /// op is anything else, it is cloned into the ClassOp and marked for erasure.
65 /// NOLINTNEXTLINE(misc-no-recursion)
66 void ExtractClassesPass::convertValue(
67  Value originalValue, OpBuilder &builder, IRMapping &mapping,
68  SmallVectorImpl<Operation *> &opsToErase) {
69  // If the Value is defined by an Operation that has already been processed,
70  // there is nothing to do.
71  if (mapping.contains(originalValue))
72  return;
73 
74  // If the Value is not defined by an Operation, there is nothing to do.
75  auto *op = originalValue.getDefiningOp();
76  if (!op)
77  return;
78 
79  // InstanceOps are handled specially, by creating ObjectOps and
80  // extracting an ObjectFieldOp. This will take care of re-using
81  // ObjectOps and ObjectFieldOps, updating the mapping, and keeping
82  // track of related ops that can be erased. The original value is
83  // mapped to the ObjectFieldOp.
84  if (auto instance = dyn_cast<InstanceOp>(op)) {
85  Value fieldValue = getOrCreateObjectFieldValue(
86  cast<OpResult>(originalValue), instance, builder, mapping, opsToErase);
87  mapping.map(originalValue, fieldValue);
88  } else {
89  // For all other ops, copy the defining op into the body, and
90  // map from the old Value to the new Value. This may need to walk
91  // property ops in order to copy them into the ClassOp, but for now
92  // only constant ops exist. Mark the property op to be erased.
93  builder.clone(*op, mapping);
94 
95  // Property defining ops should be erased after being copied over.
96  opsToErase.push_back(op);
97  }
98 }

100 /// Helper to get or create a Value from an ObjectFieldOp. This consults a cache
101 /// to avoid re-creating ObjectFieldOps. An ObjectOp is looked up or created,
102 /// and the field corresponding to the instance output is accessed. Note that
103 /// this is able to work locally with just the instance and the OpResult, and
104 /// the rest of the pass ensures the correct ClassOp is created.
105 /// NOLINTNEXTLINE(misc-no-recursion)
106 Value ExtractClassesPass::getOrCreateObjectFieldValue(
107  OpResult instanceOutput, InstanceOp instance, OpBuilder &builder,
108  IRMapping &mapping, SmallVectorImpl<Operation *> &opsToErase) {
109  // Check if this ObjectField has already been created, and return it if so.
110  auto cachedObjectValue = cachedObjectValues.find(instanceOutput);
111  if (cachedObjectValue != cachedObjectValues.end())
112  return cachedObjectValue->getSecond();
113 
114  // Get the result number for the InstanceOp output.
115  unsigned resultNum = instanceOutput.getResultNumber();
116 
117  // Get the field type.
118  Type fieldType = instance.getResult(resultNum).getType();
119 
120  // Get the ObjectOp to extract a field from.
121  ObjectOp object = getOrCreateObject(instance, builder, mapping, opsToErase);
122 
123  // Get the field path.
124  StringAttr resultName = instance.getPortName(resultNum);
125  ArrayAttr fieldPath =
126  builder.getArrayAttr(FlatSymbolRefAttr::get(resultName));
127 
128  // Construct the ObjectFieldOp.
129  auto fieldValue = builder.create<ObjectFieldOp>(instance.getLoc(), fieldType,
130  object, fieldPath);
131 
132  // Cache it for potential future lookups.
133  cachedObjectValues[instanceOutput] = fieldValue;
134 
135  return fieldValue;
136 }

138 /// Helper to get or create an ObjectOp. This consults a cache to avoid
139 /// re-creating ObjectOps. The actual parameters are computed by finding the
140 /// assignments to the InstanceOp's inputs. The property ops involved are
141 /// cloned, mapped, and marked for erasure as appropriate. The ObjectOp is then
142 /// created with the appropriate type, class name, and the actual parameters.
143 /// Note that this is able to work locally with just the instance, and the rest
144 /// of the pass ensures the correct ClassOp is created.
145 /// NOLINTNEXTLINE(misc-no-recursion)
146 ObjectOp ExtractClassesPass::getOrCreateObject(
147  InstanceOp instance, OpBuilder &builder, IRMapping &mapping,
148  SmallVectorImpl<Operation *> &opsToErase) {
149  // Check if this ObjectOp has already been created, and return it if so.
150  auto cachedObject = cachedObjects.find(instance);
151  if (cachedObject != cachedObjects.end())
152  return cachedObject->getSecond();
153 
154  // Build up the ObjectOp's actual parameters.
155  SmallVector<Value> actualParams;
156  for (size_t i = 0; i < instance.getNumResults(); ++i) {
157  // Skip outputs and non-Property inputs.
158  if (instance.getPortDirection(i) == Direction::Out)
159  continue;
160  auto result = dyn_cast<FIRRTLPropertyValue>(instance.getResult(i));
161  if (!result)
162  continue;
163 
164  // Get the assignment to the input property.
165  PropAssignOp propassign = getPropertyAssignment(result);
166 
167  // The source value will be mapped into the actual parameter.
168  Value inputValue = propassign.getSrc();
169 
170  // Convert the inputValue, if necessary.
171  convertValue(inputValue, builder, mapping, opsToErase);
172 
173  // Lookup the mapping for the input value, and use this as an actual
174  // parameter.
175  actualParams.push_back(mapping.lookup(inputValue));
176 
177  // Eagerly erase the property assign, since it is done now.
178  propassign.erase();
179  }
180 
181  // Get the Object type.
182  auto objectType =
183  ClassType::get(instance.getContext(), instance.getModuleNameAttr());
184 
185  // Get the Object's class name.
186  auto objectClass = instance.getModuleNameAttr().getAttr();
187 
188  // Construct the ObjectOp.
189  auto object = builder.create<ObjectOp>(instance.getLoc(), objectType,
190  objectClass, actualParams);
191 
192  // Cache it for potential future lookups.
193  cachedObjects[instance] = object;
194 
195  return object;
196 }

198 /// Potentially extract an OM class from a FIRRTL module which may contain
199 /// properties.
200 void ExtractClassesPass::extractClass(FModuleOp moduleOp) {
201  // Map from Values in the FModuleOp to Values in the ClassOp.
202  IRMapping mapping;
203 
204  // Remember ports and operations to clean up when done.
205  portsToErase[moduleOp] = llvm::BitVector(moduleOp.getNumPorts());
206  SmallVector<Operation *> opsToErase;
207 
208  // Collect information about input and output properties. Mark property ports
209  // to be erased.
210  SmallVector<Property> inputProperties;
211  SmallVector<Property> outputProperties;
212  for (auto [index, port] : llvm::enumerate(moduleOp.getPorts())) {
213  if (!isa<PropertyType>(port.type))
214  continue;
215 
216  portsToErase[moduleOp].set(index);
217 
218  if (port.isInput())
219  inputProperties.push_back({index, port.name, port.type, port.loc});
220 
221  if (port.isOutput())
222  outputProperties.push_back({index, port.name, port.type, port.loc});
223  }
224 
225  // If the FModuleOp has no properties, nothing to do.
226  if (inputProperties.empty() && outputProperties.empty())
227  return;
228 
229  OpBuilder builder = OpBuilder::atBlockEnd(getOperation().getBody(0));
230 
231  // Collect the parameter names from input properties.
232  SmallVector<StringRef> formalParamNames;
233  for (auto inputProperty : inputProperties)
234  formalParamNames.push_back(inputProperty.name);
235 
236  // Construct the ClassOp with the FModuleOp name and parameter names.
237  auto classOp = builder.create<om::ClassOp>(
238  moduleOp.getLoc(), moduleOp.getName(), formalParamNames);
239 
240  // Construct the ClassOp body with block arguments for each input property,
241  // updating the mapping to map from the input property to the block argument.
242  Block *classBody = &classOp.getRegion().emplaceBlock();
243  for (auto inputProperty : inputProperties) {
244  BlockArgument parameterValue =
245  classBody->addArgument(inputProperty.type, inputProperty.loc);
246  BlockArgument inputValue = moduleOp.getArgument(inputProperty.index);
247  mapping.map(inputValue, parameterValue);
248  }
249 
250  // Construct ClassFieldOps for each output property.
251  builder.setInsertionPointToStart(classBody);
252  for (auto outputProperty : outputProperties) {
253  // Get the Value driven to the property to use for this ClassFieldOp.
254  auto outputValue =
255  cast<FIRRTLPropertyValue>(moduleOp.getArgument(outputProperty.index));
256  Value originalValue = getDriverFromConnect(outputValue);
257 
258  // Convert the inputValue, if necessary.
259  convertValue(originalValue, builder, mapping, opsToErase);
260 
261  // Create the ClassFieldOp using the mapping to find the appropriate Value.
262  Value fieldValue = mapping.lookup(originalValue);
263  builder.create<ClassFieldOp>(originalValue.getLoc(), outputProperty.name,
264  fieldValue);
265 
266  // Eagerly erase the property assign, since it is done now.
267  getPropertyAssignment(outputValue).erase();
268  }
269 
270  // Clean up the FModuleOp by removing property ports and operations. This
271  // first erases opsToErase in the order they were added, so property
272  // assignments are erased before value defining ops. Then it erases ports.
273  for (auto *op : opsToErase)
274  op->erase();
275  moduleOp.erasePorts(portsToErase[moduleOp]);
276 }

278 /// Clean up InstanceOps of any FModuleOps with properties.
279 void ExtractClassesPass::updateInstances(FModuleOp moduleOp) {
280  OpBuilder builder(&getContext());
281  const llvm::BitVector &modulePortsToErase = portsToErase[moduleOp];
282  InstanceGraphNode *instanceGraphNode = instanceGraph->lookup(moduleOp);
283 
284  // If there are no ports to erase, nothing to do.
285  if (!modulePortsToErase.empty() && !modulePortsToErase.any())
286  return;
287 
288  // Clean up instances of the FModuleOp.
289  for (InstanceRecord *node :
290  llvm::make_early_inc_range(instanceGraphNode->uses())) {
291  // Get the original InstanceOp.
292  InstanceOp oldInstance = cast<InstanceOp>(node->getInstance());
293  builder.setInsertionPointAfter(oldInstance);
294 
295  // If some but not all ports are properties, create a new instance without
296  // the property pins.
297  if (!modulePortsToErase.all()) {
298  InstanceOp newInstance =
299  oldInstance.erasePorts(builder, portsToErase[moduleOp]);
300  instanceGraph->replaceInstance(oldInstance, newInstance);
301  }
302 
303  // Clean up uses of property pins. This amounts to erasing property
304  // assignments for now.
305  for (int propertyIndex : modulePortsToErase.set_bits()) {
306  for (Operation *user : llvm::make_early_inc_range(
307  oldInstance.getResult(propertyIndex).getUsers())) {
308  assert(isa<FConnectLike>(user) &&
309  "expected property pins to be used in property assignments");
310  user->erase();
311  }
312  }
313 
314  // Erase the original instance.
315  node->erase();
316  oldInstance.erase();
317  }
318 
319  // If all ports are properties, remove the FModuleOp completely.
320  if (!modulePortsToErase.empty() && modulePortsToErase.all()) {
321  instanceGraph->erase(instanceGraphNode);
322  moduleOp.erase();
323  }
324 }

326 /// Extract OM classes from FIRRTL modules with properties.
327 void ExtractClassesPass::runOnOperation() {
328  // Get the CircuitOp.
329  auto circuits = getOperation().getOps<CircuitOp>();
330  if (circuits.empty())
331  return;
332  CircuitOp circuit = *circuits.begin();
333 
334  // Get the FIRRTL instance graph.
335  instanceGraph = &getAnalysis<InstanceGraph>();
336 
337  // Walk all FModuleOps to potentially extract an OM class if the FModuleOp
338  // contains properties.
339  for (auto moduleOp : llvm::make_early_inc_range(circuit.getOps<FModuleOp>()))
340  extractClass(moduleOp);
341 
342  // Clean up InstanceOps of any FModuleOps with properties. This is done after
343  // the classes are extracted to avoid extra bookeeping as InstanceOps are
344  // cleaned up.
345  for (auto moduleOp : llvm::make_early_inc_range(circuit.getOps<FModuleOp>()))
346  updateInstances(moduleOp);
347 
348  // Mark analyses preserved, since we keep the instance graph up to date.
349  markAllAnalysesPreserved();
350 
351  // Reset pass state.
352  instanceGraph = nullptr;
353 }
