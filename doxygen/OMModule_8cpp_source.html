24 /// Provides an Object class by simply wrapping the OMObject CAPI.
25 struct Object {
26  // Instantiate an Object with a reference to the underlying OMObject.
27  Object(OMObject object) : object(object) {}
28  Object(const Object &object) : object(object.object) {}
29 
30  /// Get the Type from an Object, which will be a ClassType.
31  MlirType getType() { return omEvaluatorObjectGetType(object); }
32 
33  // Get a field from the Object, using pybind's support for variant to return a
34  // Python object that is either an Object or Attribute.
35  std::variant<Object, MlirAttribute> getField(const std::string &name) {
36  // Wrap the requested field name in an attribute.
37  MlirContext context = mlirTypeGetContext(omEvaluatorObjectGetType(object));
38  MlirStringRef cName = mlirStringRefCreateFromCString(name.c_str());
39  MlirAttribute nameAttr = mlirStringAttrGet(context, cName);
40 
41  // Get the field's ObjectValue via the CAPI.
42  OMObjectValue result = omEvaluatorObjectGetField(object, nameAttr);
43 
44  // If the ObjectValue is null, something failed. Diagnostic handling is
45  // implemented in pure Python, so nothing to do here besides throwing an
46  // error to halt execution.
47  if (omEvaluatorObjectValueIsNull(result))
48  throw py::value_error("unable to get field, see previous error(s)");
49 
50  // If the field was an Object, return a new Object.
51  if (omEvaluatorObjectValueIsAObject(result))
52  return Object(omEvaluatorObjectValueGetObject(result));
53 
54  // If the field was a primitive, return the Attribute.
55  assert(omEvaluatorObjectValueIsAPrimitive(result));
56  return omEvaluatorObjectValueGetPrimitive(result);
57  }
58 
59  // Get a list with the names of all the fields in the Object.
60  std::vector<std::string> getFieldNames() {
61  MlirAttribute fieldNames = omEvaluatorObjectGetFieldNames(object);
62  intptr_t numFieldNames = mlirArrayAttrGetNumElements(fieldNames);
63 
64  std::vector<std::string> pyFieldNames;
65  for (intptr_t i = 0; i < numFieldNames; ++i) {
66  MlirAttribute fieldName = mlirArrayAttrGetElement(fieldNames, i);
67  MlirStringRef fieldNameStr = mlirStringAttrGetValue(fieldName);
68  pyFieldNames.emplace_back(fieldNameStr.data, fieldNameStr.length);
69  }
70 
71  return pyFieldNames;
72  }
73 
74 private:
75  // The underlying CAPI OMObject.
76  OMObject object;
77 };

86  std::vector<MlirAttribute> actualParams) {
87  // Instantiate the Object via the CAPI.
88  OMObject result = omEvaluatorInstantiate(

90

95  throw py::value_error(

97 
98  // Return a new Object.
99  return Object(result);
100  }
101

103  MlirModule getModule() { return omEvaluatorGetModule(evaluator); }

108 };

136 /// Populate the OM Python module.
137 void circt::python::populateDialectOMSubmodule(py::module &m) {
138  m.doc() = "OM dialect Python native extension";
139 
140  // Add the Evaluator class definition.
141  py::class_<Evaluator>(m, "Evaluator")
142  .def(py::init<MlirModule>(), py::arg("module"))
143  .def("instantiate", &Evaluator::instantiate, "Instantiate an Object",
144  py::arg("class_name"), py::arg("actual_params"))
145  .def_property_readonly("module", &Evaluator::getModule,
146  "The Module the Evaluator is built from");
147 
148  // Add the Object class definition.
149  py::class_<Object>(m, "Object")
150  .def(py::init<Object>(), py::arg("object"))
151  .def("__getattr__", &Object::getField, "Get a field from an Object",
152  py::arg("name"))
153  .def_property_readonly("field_names", &Object::getFieldNames,
154  "Get field names from an Object")
155  .def_property_readonly("type", &Object::getType,
156  "The Type of the Object");
157 
158  // Add the ReferenceAttr definition
159  mlir_attribute_subclass(m, "ReferenceAttr", omAttrIsAReferenceAttr)
160  .def_property_readonly("inner_ref", [](MlirAttribute self) {
161  return omReferenceAttrGetInnerRef(self);
162  });
163 
164  // Add the OMListAttr definition
165  mlir_attribute_subclass(m, "ListAttr", omAttrIsAListAttr)
166  .def("__getitem__", &omListAttrGetElement)
167  .def("__len__", &omListAttrGetElement)
168  .def("__iter__",
169  [](MlirAttribute arr) { return PyListAttrIterator(arr); });
170  PyListAttrIterator::bind(m);
171 
172  // Add the ClassType class definition.
173  mlir_type_subclass(m, "ClassType", omTypeIsAClassType);
174 }
