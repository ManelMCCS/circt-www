34 struct FirMemConfig {
35  size_t numReadPorts = 0;
36  size_t numWritePorts = 0;
37  size_t numReadWritePorts = 0;
38  size_t dataWidth = 0;
39  size_t depth = 0;
40  size_t readLatency = 0;
41  size_t writeLatency = 0;
42  size_t maskBits = 0;
43  RUW readUnderWrite = RUW::Undefined;
44  WUW writeUnderWrite = WUW::Undefined;
45  SmallVector<int32_t, 1> writeClockIDs;
46  StringRef initFilename;
47  bool initIsBinary = false;
48  bool initIsInline = false;
49  Attribute outputFile;
50  StringRef prefix;
51 
52  llvm::hash_code hashValue() const {
53  return llvm::hash_combine(numReadPorts, numWritePorts, numReadWritePorts,
54  dataWidth, depth, readLatency, writeLatency,
55  maskBits, readUnderWrite, writeUnderWrite,
56  initFilename, initIsBinary, initIsInline,
57  outputFile, prefix) ^
58  llvm::hash_combine_range(writeClockIDs.begin(), writeClockIDs.end());
59  }
60 
61  auto getTuple() const {
62  return std::make_tuple(numReadPorts, numWritePorts, numReadWritePorts,
63  dataWidth, depth, readLatency, writeLatency,
64  maskBits, readUnderWrite, writeUnderWrite,
65  writeClockIDs, initFilename, initIsBinary,
66  initIsInline, outputFile, prefix);
67  }
68 
69  bool operator==(const FirMemConfig &other) const {
70  return getTuple() == other.getTuple();
71  }

128 void LowerFirMemPass::runOnOperation() {
129  // Gather all HW modules. We'll parallelize over them.
130  SmallVector<HWModuleOp> modules;
131  getOperation().walk([&](HWModuleOp op) {
132  modules.push_back(op);
133  return WalkResult::skip();
134  });
135  LLVM_DEBUG(llvm::dbgs() << "Lowering memories in " << modules.size()
136  << " modules\n");
137 
138  // Gather all `FirMemOp`s in the HW modules and group them by configuration.
139  auto uniqueMems = collectMemories(modules);
140  LLVM_DEBUG(llvm::dbgs() << "Found " << uniqueMems.size()
141  << " unique memory congiurations\n");
142  if (uniqueMems.empty()) {
143  markAllAnalysesPreserved();
144  return;
145  }
146 
147  // Create the `HWModuleGeneratedOp`s for each unique configuration. The result
148  // is a vector of the same size as `uniqueMems`, with a `HWModuleGeneratedOp`
149  // for every unique memory configuration.
150  auto genOps = createMemoryModules(uniqueMems);
151 
152  // Group the list of memories that we need to update per HW module. This will
153  // allow us to parallelize across HW modules.
154  MapVector<
155  HWModuleOp,
156  SmallVector<std::tuple<FirMemConfig *, HWModuleGeneratedOp, FirMemOp>>>
157  memsToLowerByModule;
158 
159  for (auto [config, genOp] : llvm::zip(uniqueMems, genOps))
160  for (auto memOp : config.second)
161  memsToLowerByModule[memOp->getParentOfType<HWModuleOp>()].push_back(
162  {&config.first, genOp, memOp});
163 
164  // Replace all `FirMemOp`s with instances of the generated module.
165  if (getContext().isMultithreadingEnabled()) {
166  llvm::parallelForEach(memsToLowerByModule, [&](auto pair) {
167  lowerMemoriesInModule(pair.first, pair.second);
168  });
169  } else {
170  for (auto [module, mems] : memsToLowerByModule)
171  lowerMemoriesInModule(module, mems);
172  }
173 }

175 /// Collect the memories in a list of HW modules.
176 LowerFirMemPass::UniqueConfigs
177 LowerFirMemPass::collectMemories(ArrayRef<HWModuleOp> modules) {
178  // For each module in the list populate a separate vector of `FirMemOp`s in
179  // that module. This allows for the traversal of the HW modules to be
180  // parallelized.
181  using ModuleMemories = SmallVector<std::pair<FirMemConfig, FirMemOp>, 0>;
182  SmallVector<ModuleMemories> memories(modules.size());
183 
184  auto collect = [&](HWModuleOp module, ModuleMemories &memories) {
185  // TODO: Check if this module is in the DUT hierarchy.
186  // bool isInDut = state.isInDUT(module);
187  module.walk([&](seq::FirMemOp op) {
188  memories.push_back({collectMemory(op), op});
189  });
190  };
191 
192  if (getContext().isMultithreadingEnabled()) {
193  llvm::parallelFor(0, modules.size(),
194  [&](auto idx) { collect(modules[idx], memories[idx]); });
195  } else {
196  for (auto [module, moduleMemories] : llvm::zip(modules, memories))
197  collect(module, moduleMemories);
198  }
199 
200  // Group the gathered memories by unique `FirMemConfig` details.
201  MapVector<FirMemConfig, SmallVector<FirMemOp, 1>> grouped;
202  for (auto [module, moduleMemories] : llvm::zip(modules, memories))
203  for (auto [summary, memOp] : moduleMemories)
204  grouped[summary].push_back(memOp);
205 
206  return grouped.takeVector();
207 }

209 /// Trace a value through wires to its original definition.
210 static Value lookThroughWires(Value value) {
211  while (value) {
212  if (auto wireOp = value.getDefiningOp<WireOp>()) {
213  value = wireOp.getInput();
214  continue;
215  }
216  break;
217  }
218  return value;
219 }

221 /// Determine the exact parametrization of the memory that should be generated
222 /// for a given `FirMemOp`.
223 FirMemConfig LowerFirMemPass::collectMemory(FirMemOp op) {
224  FirMemConfig cfg;
225  cfg.dataWidth = op.getType().getWidth();
226  cfg.depth = op.getType().getDepth();
227  cfg.readLatency = op.getReadLatency();
228  cfg.writeLatency = op.getWriteLatency();
229  cfg.maskBits = op.getType().getMaskWidth().value_or(1);
230  cfg.readUnderWrite = op.getRuw();
231  cfg.writeUnderWrite = op.getWuw();
232  if (auto init = op.getInitAttr()) {
233  cfg.initFilename = init.getFilename();
234  cfg.initIsBinary = init.getIsBinary();
235  cfg.initIsInline = init.getIsInline();
236  }
237  cfg.outputFile = op.getOutputFileAttr();
238  if (auto prefix = op.getPrefixAttr())
239  cfg.prefix = prefix.getValue();
240  // TODO: Handle modName (maybe not?)
241  // TODO: Handle groupID (maybe not?)
242 
243  // Count the read, write, and read-write ports, and identify the clocks
244  // driving the write ports.
245  SmallDenseMap<Value, unsigned> clockValues;
246  for (auto *user : op->getUsers()) {
247  if (isa<FirMemReadOp>(user))
248  ++cfg.numReadPorts;
249  else if (isa<FirMemWriteOp>(user))
250  ++cfg.numWritePorts;
251  else if (isa<FirMemReadWriteOp>(user))
252  ++cfg.numReadWritePorts;
253 
254  // Assign IDs to the values used as clock. This allows later passes to
255  // easily detect which clocks are effectively driven by the same value.
256  if (isa<FirMemWriteOp, FirMemReadWriteOp>(user)) {
257  auto clock = lookThroughWires(user->getOperand(2));
258  cfg.writeClockIDs.push_back(
259  clockValues.insert({clock, clockValues.size()}).first->second);
260  }
261  }
262 
263  return cfg;
264 }

266 /// Create the `HWModuleGeneratedOp` for a list of memory parametrizations.
267 SmallVector<HWModuleGeneratedOp>
268 LowerFirMemPass::createMemoryModules(MutableArrayRef<UniqueConfig> configs) {
269  ModuleOp circuit = getOperation();
270 
271  // Create or re-use the generator schema.
272  hw::HWGeneratorSchemaOp schemaOp;
273  for (auto op : circuit.getOps<hw::HWGeneratorSchemaOp>()) {
274  if (op.getDescriptor() == "FIRRTL_Memory") {
275  schemaOp = op;
276  break;
277  }
278  }
279  if (!schemaOp) {
280  auto builder = OpBuilder::atBlockBegin(getOperation().getBody());
281  std::array<StringRef, 14> schemaFields = {
282  "depth", "numReadPorts",
283  "numWritePorts", "numReadWritePorts",
284  "readLatency", "writeLatency",
285  "width", "maskGran",
286  "readUnderWrite", "writeUnderWrite",
287  "writeClockIDs", "initFilename",
288  "initIsBinary", "initIsInline"};
289  schemaOp = builder.create<hw::HWGeneratorSchemaOp>(
290  getOperation().getLoc(), "FIRRTLMem", "FIRRTL_Memory",
291  builder.getStrArrayAttr(schemaFields));
292  }
293  auto schemaSymRef = FlatSymbolRefAttr::get(schemaOp);
294 
295  // Determine the insertion point for each of the memory modules. We basically
296  // put them ahead of the first module that instantiates that memory. Do this
297  // here in one go such that the `isBeforeInBlock` calls don't have to
298  // re-enumerate the entire IR every time we insert one of the memory modules.
299  SmallVector<Operation *> insertionPoints;
300  insertionPoints.reserve(configs.size());
301  for (auto &config : configs) {
302  Operation *op = nullptr;
303  for (auto memOp : config.second)
304  if (auto parent = memOp->getParentOfType<HWModuleOp>())
305  if (!op || parent->isBeforeInBlock(op))
306  op = parent;
307  insertionPoints.push_back(op);
308  }
309 
310  // Create the individual memory modules.
311  SymbolCache symbolCache;
312  symbolCache.addDefinitions(getOperation());
313  Namespace globalNamespace;
314  globalNamespace.add(symbolCache);
315 
316  SmallVector<HWModuleGeneratedOp> genOps;
317  genOps.reserve(configs.size());
318  for (auto [config, insertBefore] : llvm::zip(configs, insertionPoints)) {
319  OpBuilder builder(circuit.getContext());
320  builder.setInsertionPoint(insertBefore);
321  genOps.push_back(
322  createMemoryModule(config, builder, schemaSymRef, globalNamespace));
323  }
324 
325  return genOps;
326 }

328 /// Create the `HWModuleGeneratedOp` for a single memory parametrization.
329 HWModuleGeneratedOp
330 LowerFirMemPass::createMemoryModule(UniqueConfig &config, OpBuilder &builder,
331  FlatSymbolRefAttr schemaSymRef,
332  Namespace &globalNamespace) {
333  const auto &mem = config.first;
334  auto &memOps = config.second;
335 
336  // Pick a name for the memory. Honor the optional prefix and try to include
337  // the common part of the names of the memory instances that use this
338  // configuration. The resulting name is of the form:
339  //
340  // <prefix>_<commonName>_<depth>x<width>
341  //
342  StringRef baseName = "";
343  bool firstFound = false;
344  for (auto memOp : memOps) {
345  if (auto memName = memOp.getName()) {
346  if (!firstFound) {
347  baseName = *memName;
348  firstFound = true;
349  continue;
350  }
351  unsigned idx = 0;
352  for (; idx < memName->size() && idx < baseName.size(); ++idx)
353  if ((*memName)[idx] != baseName[idx])
354  break;
355  baseName = baseName.take_front(idx);
356  }
357  }
358  baseName = baseName.rtrim('_');
359 
360  SmallString<32> nameBuffer;
361  nameBuffer += mem.prefix;
362  if (!baseName.empty()) {
363  nameBuffer += baseName;
364  } else {
365  nameBuffer += "mem";
366  }
367  nameBuffer += "_";
368  (Twine(mem.depth) + "x" + Twine(mem.dataWidth)).toVector(nameBuffer);
369  auto name = builder.getStringAttr(globalNamespace.newName(nameBuffer));
370 
371  LLVM_DEBUG(llvm::dbgs() << "Creating " << name << " for " << mem.depth
372  << " x " << mem.dataWidth << " memory\n");
373 
374  bool withMask = mem.maskBits > 1;
375  SmallVector<hw::PortInfo> ports;
376 
377  // Common types used for memory ports.
378  Type bitType = IntegerType::get(&getContext(), 1);
379  Type dataType =
380  IntegerType::get(&getContext(), std::max((size_t)1, mem.dataWidth));
381  Type maskType = IntegerType::get(&getContext(), mem.maskBits);
382  Type addrType = IntegerType::get(&getContext(),
383  std::max(1U, llvm::Log2_64_Ceil(mem.depth)));
384 
385  // Helper to add an input port.
386  size_t inputIdx = 0;
387  auto addInput = [&](StringRef prefix, size_t idx, StringRef suffix,
388  Type type) {
389  ports.push_back({{builder.getStringAttr(prefix + Twine(idx) + suffix), type,
390  ModulePort::Direction::Input},
391  inputIdx++});
392  };
393 
394  // Helper to add an output port.
395  size_t outputIdx = 0;
396  auto addOutput = [&](StringRef prefix, size_t idx, StringRef suffix,
397  Type type) {
398  ports.push_back({{builder.getStringAttr(prefix + Twine(idx) + suffix), type,
399  ModulePort::Direction::Output},
400  outputIdx++});
401  };
402 
403  // Helper to add the ports common to read, read-write, and write ports.
404  auto addCommonPorts = [&](StringRef prefix, size_t idx) {
405  addInput(prefix, idx, "_addr", addrType);
406  addInput(prefix, idx, "_en", bitType);
407  addInput(prefix, idx, "_clk", bitType);
408  };
409 
410  // Add the read ports.
411  for (size_t i = 0, e = mem.numReadPorts; i != e; ++i) {
412  addCommonPorts("R", i);
413  addOutput("R", i, "_data", dataType);
414  }
415 
416  // Add the read-write ports.
417  for (size_t i = 0, e = mem.numReadWritePorts; i != e; ++i) {
418  addCommonPorts("RW", i);
419  addInput("RW", i, "_wmode", bitType);
420  addInput("RW", i, "_wdata", dataType);
421  addOutput("RW", i, "_rdata", dataType);
422  if (withMask)
423  addInput("RW", i, "_wmask", maskType);
424  }
425 
426  // Add the write ports.
427  for (size_t i = 0, e = mem.numWritePorts; i != e; ++i) {
428  addCommonPorts("W", i);
429  addInput("W", i, "_data", dataType);
430  if (withMask)
431  addInput("W", i, "_mask", maskType);
432  }
433 
434  // Mask granularity is the number of data bits that each mask bit can
435  // guard. By default it is equal to the data bitwidth.
436  auto genAttr = [&](StringRef name, Attribute attr) {
437  return builder.getNamedAttr(name, attr);
438  };
439  auto genAttrUI32 = [&](StringRef name, uint32_t value) {
440  return genAttr(name, builder.getUI32IntegerAttr(value));
441  };
442  NamedAttribute genAttrs[] = {
443  genAttr("depth", builder.getI64IntegerAttr(mem.depth)),
444  genAttrUI32("numReadPorts", mem.numReadPorts),
445  genAttrUI32("numWritePorts", mem.numWritePorts),
446  genAttrUI32("numReadWritePorts", mem.numReadWritePorts),
447  genAttrUI32("readLatency", mem.readLatency),
448  genAttrUI32("writeLatency", mem.writeLatency),
449  genAttrUI32("width", mem.dataWidth),
450  genAttrUI32("maskGran", mem.dataWidth / mem.maskBits),
451  genAttr("readUnderWrite",
452  seq::RUWAttr::get(builder.getContext(), mem.readUnderWrite)),
453  genAttr("writeUnderWrite",
454  seq::WUWAttr::get(builder.getContext(), mem.writeUnderWrite)),
455  genAttr("writeClockIDs", builder.getI32ArrayAttr(mem.writeClockIDs)),
456  genAttr("initFilename", builder.getStringAttr(mem.initFilename)),
457  genAttr("initIsBinary", builder.getBoolAttr(mem.initIsBinary)),
458  genAttr("initIsInline", builder.getBoolAttr(mem.initIsInline))};
459 
460  // Combine the locations of all actual `FirMemOp`s to be the location of the
461  // generated memory.
462  Location loc = memOps.front().getLoc();
463  if (memOps.size() > 1) {
464  SmallVector<Location> locs;
465  for (auto memOp : memOps)
466  locs.push_back(memOp.getLoc());
467  loc = FusedLoc::get(&getContext(), locs);
468  }
469 
470  // Create the module.
471  auto genOp = builder.create<hw::HWModuleGeneratedOp>(
472  loc, schemaSymRef, name, ports, StringRef{}, ArrayAttr{}, genAttrs);
473  if (mem.outputFile)
474  genOp->setAttr("output_file", mem.outputFile);
475 
476  return genOp;
477 }

479 /// Replace all `FirMemOp`s in an HW module with an instance of the
480 /// corresponding generated module.
481 void LowerFirMemPass::lowerMemoriesInModule(
482  HWModuleOp module,
483  ArrayRef<std::tuple<FirMemConfig *, HWModuleGeneratedOp, FirMemOp>> mems) {
484  LLVM_DEBUG(llvm::dbgs() << "Lowering " << mems.size() << " memories in "
485  << module.getName() << "\n");
486 
487  hw::ConstantOp constOneOp;
488  auto constOne = [&] {
489  if (!constOneOp) {
490  auto builder = OpBuilder::atBlockBegin(module.getBodyBlock());
491  constOneOp = builder.create<hw::ConstantOp>(module.getLoc(),
492  builder.getI1Type(), 1);
493  }
494  return constOneOp;
495  };
496  auto valueOrOne = [&](Value value) { return value ? value : constOne(); };
497 
498  for (auto [config, genOp, memOp] : mems) {
499  LLVM_DEBUG(llvm::dbgs() << "- Lowering " << memOp.getName() << "\n");
500  SmallVector<Value> inputs;
501  SmallVector<Value> outputs;
502 
503  auto addInput = [&](Value value) { inputs.push_back(value); };
504  auto addOutput = [&](Value value) { outputs.push_back(value); };
505 
506  // Add the read ports.
507  for (auto *op : memOp->getUsers()) {
508  auto port = dyn_cast<FirMemReadOp>(op);
509  if (!port)
510  continue;
511  addInput(port.getAddress());
512  addInput(valueOrOne(port.getEnable()));
513  addInput(port.getClock());
514  addOutput(port.getData());
515  }
516 
517  // Add the read-write ports.
518  for (auto *op : memOp->getUsers()) {
519  auto port = dyn_cast<FirMemReadWriteOp>(op);
520  if (!port)
521  continue;
522  addInput(port.getAddress());
523  addInput(valueOrOne(port.getEnable()));
524  addInput(port.getClock());
525  addInput(port.getMode());
526  addInput(port.getWriteData());
527  addOutput(port.getReadData());
528  if (config->maskBits > 1)
529  addInput(valueOrOne(port.getMask()));
530  }
531 
532  // Add the write ports.
533  for (auto *op : memOp->getUsers()) {
534  auto port = dyn_cast<FirMemWriteOp>(op);
535  if (!port)
536  continue;
537  addInput(port.getAddress());
538  addInput(valueOrOne(port.getEnable()));
539  addInput(port.getClock());
540  addInput(port.getData());
541  if (config->maskBits > 1)
542  addInput(valueOrOne(port.getMask()));
543  }
544 
545  // Create the module instance.
546  StringRef memName = "mem";
547  if (auto name = memOp.getName(); name && !name->empty())
548  memName = *name;
549  ImplicitLocOpBuilder builder(memOp.getLoc(), memOp);
550  auto instOp = builder.create<hw::InstanceOp>(
551  genOp, builder.getStringAttr(memName + "_ext"), inputs, ArrayAttr{},
552  memOp.getInnerSymAttr());
553  for (auto [oldOutput, newOutput] : llvm::zip(outputs, instOp.getResults()))
554  oldOutput.replaceAllUsesWith(newOutput);
555 
556  // Carry attributes over from the `FirMemOp` to the `InstanceOp`.
557  auto defaultAttrNames = memOp.getAttributeNames();
558  for (auto namedAttr : memOp->getAttrs())
559  if (!llvm::is_contained(defaultAttrNames, namedAttr.getName()))
560  instOp->setAttr(namedAttr.getName(), namedAttr.getValue());
561 
562  // Get rid of the `FirMemOp`.
563  for (auto *user : llvm::make_early_inc_range(memOp->getUsers()))
564  user->erase();
565  memOp.erase();
566  }
567 }
