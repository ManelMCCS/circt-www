79 /// The FIRRTL specification version.
80 struct FIRVersion {
81  uint32_t major, minor, patch;
82 
83  /// Three way compare of one FIRRTL version with another FIRRTL version.
84  /// Return 1 if the first version is greater than the second version, -1 if
85  /// the first version is less than the second version, and 0 if the versions
86  /// are equal.
87  static int compare(const FIRVersion &a, const FIRVersion &b) {
88  if (a.major > b.major)
89  return 1;
90  if (a.major < b.major)
91  return -1;
92  if (a.minor > b.minor)
93  return 1;
94  if (a.minor < b.minor)
95  return -1;
96  if (a.patch > b.patch)
97  return 1;
98  if (a.patch < b.patch)
99  return -1;
100  return 0;
101  }
102 
103  static FIRVersion minimumFIRVersion() { return {0, 2, 0}; }
104 
105  static FIRVersion defaultFIRVersion() { return {1, 0, 0}; }
106 
107 }; // namespace firrtl
