25 /// This holds the name, type, direction of a module's ports
26 struct PortInfo : public ModulePort {
27  /// This is the argument index or the result index depending on the direction.
28  /// "0" for an output means the first output, "0" for a in/inout means the
29  /// first argument.
30  size_t argNum = ~0U;
31 
32  /// The optional symbol for this port.
33  InnerSymAttr sym = {};
34  LocationAttr loc = {};
35 
36  StringRef getName() const { return name.getValue(); }
37  bool isInput() const { return dir == ModulePort::Direction::Input; }
38  bool isOutput() const { return dir == ModulePort::Direction::Output; }
39  bool isInOut() const { return dir == ModulePort::Direction::InOut; }
40 
41  /// Return a unique numeric identifier for this port.
42  ssize_t getId() const { return isOutput() ? argNum : (-1 - argNum); };
43 };

45 /// This holds a decoded list of input/inout and output ports for a module or
46 /// instance.
47 struct ModulePortInfo {
48  explicit ModulePortInfo(ArrayRef<PortInfo> inputs,
49  ArrayRef<PortInfo> outputs) {
50  ports.insert(ports.end(), inputs.begin(), inputs.end());
51  numInputs = ports.size();
52  ports.insert(ports.end(), outputs.begin(), outputs.end());
53  }
54 
55  explicit ModulePortInfo(ArrayRef<PortInfo> mergedPorts)
56  : ports(mergedPorts.begin(), mergedPorts.end()), numInputs(0) {
57  std::stable_sort(ports.begin(), ports.end(),
58  [](const PortInfo &lhs, const PortInfo &rhs) {
59  return !lhs.isOutput() && rhs.isOutput();
60  });
61  while (numInputs < ports.size() && !ports[numInputs].isOutput())
62  ++numInputs;
63  }
64 
65  using iterator = SmallVector<PortInfo>::iterator;
66  using const_iterator = SmallVector<PortInfo>::const_iterator;
67  iterator begin_input() { return ports.begin(); }
68  iterator end_input() { return ports.begin() + numInputs; }
69  const_iterator begin_input() const { return ports.begin(); }
70  const_iterator end_input() const { return ports.begin() + numInputs; }
71  iterator begin_output() { return end_input(); }
72  iterator end_output() { return ports.end(); }
73  const_iterator begin_output() const { return end_input(); }
74  const_iterator end_output() const { return ports.end(); }
75  iterator begin() { return ports.begin(); }
76  iterator end() { return ports.end(); }
77  const_iterator begin() const { return ports.begin(); }
78  const_iterator end() const { return ports.end(); }
79 
80  llvm::iterator_range<iterator> all() { return {begin(), end()}; }
81  llvm::iterator_range<const_iterator> all() const { return {begin(), end()}; }
82  llvm::iterator_range<iterator> inputs() {
83  return {begin_input(), end_input()};
84  }
85  llvm::iterator_range<const_iterator> inputs() const {
86  return {begin_input(), end_input()};
87  }
88  llvm::iterator_range<iterator> outputs() {
89  return {begin_output(), end_output()};
90  }
91  llvm::iterator_range<const_iterator> outputs() const {
92  return {begin_output(), end_output()};
93  }
94 
95  size_t size() const { return ports.size(); }
96  size_t sizeInputs() const { return numInputs; }
97  size_t sizeOutputs() const { return ports.size() - numInputs; }
98 
99  PortInfo &at(size_t idx) { return ports[idx]; }
100  PortInfo &atInput(size_t idx) { return ports[idx]; }
101  PortInfo &atOutput(size_t idx) { return ports[idx + numInputs]; }
102 
103  const PortInfo &at(size_t idx) const { return ports[idx]; }
104  const PortInfo &atInput(size_t idx) const { return ports[idx]; }
105  const PortInfo &atOutput(size_t idx) const { return ports[idx + numInputs]; }
106 
107  void eraseInput(size_t idx) {
108  assert(numInputs);
109  assert(idx < numInputs);
110  ports.erase(ports.begin() + idx);
111  --numInputs;
112  }
113 
114 private:
115  /// This contains a list of all ports. Input first.
116  SmallVector<PortInfo> ports;
117 
118  /// This is the index of the first output port
119  size_t numInputs;
120 };

122 // This provides capability for looking up port indices based on port names.
123 struct ModulePortLookupInfo {
124  FailureOr<unsigned>
125  lookupPortIndex(const llvm::DenseMap<StringAttr, unsigned> &portMap,
126  StringAttr name) const {
127  auto it = portMap.find(name);
128  if (it == portMap.end())
129  return failure();
130  return it->second;
131  }
132 
133 public:
134  explicit ModulePortLookupInfo(MLIRContext *ctx,
135  const ModulePortInfo &portInfo)
136  : ctx(ctx) {
137  for (auto &in : portInfo.inputs())
138  inputPortMap[in.name] = in.argNum;
139 
140  for (auto &out : portInfo.outputs())
141  outputPortMap[out.name] = out.argNum;
142  }
143 
144  // Return the index of the input port with the specified name.
145  FailureOr<unsigned> getInputPortIndex(StringAttr name) const {
146  return lookupPortIndex(inputPortMap, name);
147  }
148 
149  // Return the index of the output port with the specified name.
150  FailureOr<unsigned> getOutputPortIndex(StringAttr name) const {
151  return lookupPortIndex(outputPortMap, name);
152  }
153 
154  FailureOr<unsigned> getInputPortIndex(StringRef name) const {
155  return getInputPortIndex(StringAttr::get(ctx, name));
156  }
157 
158  FailureOr<unsigned> getOutputPortIndex(StringRef name) const {
159  return getOutputPortIndex(StringAttr::get(ctx, name));
160  }
161 
162 private:
163  llvm::DenseMap<StringAttr, unsigned> inputPortMap;
164  llvm::DenseMap<StringAttr, unsigned> outputPortMap;
165  MLIRContext *ctx;
166 };

182 /// This trait is for operations that define a scope for resolving InnerRef's,
183 /// and provides verification for InnerRef users (via InnerRefUserOpInterface).
184 template <typename ConcreteType>
185 class InnerRefNamespace : public TraitBase<ConcreteType, InnerRefNamespace> {
186 public:
187  static LogicalResult verifyRegionTrait(Operation *op) {
188  static_assert(
189  ConcreteType::template hasTrait<::mlir::OpTrait::SymbolTable>(),
190  "expected operation to be a SymbolTable");
191 
192  if (op->getNumRegions() != 1)
193  return op->emitError("expected operation to have a single region");
194  if (!op->getRegion(0).hasOneBlock())
195  return op->emitError("expected operation to have a single block");
196 
197  // Verify all InnerSymbolTable's and InnerRef users.
198  return ::circt::hw::detail::verifyInnerRefNamespace(op);
199  }
200 };

202 /// A trait for inner symbol table functionality on an operation.
203 template <typename ConcreteType>
204 class InnerSymbolTable : public TraitBase<ConcreteType, InnerSymbolTable> {
205 public:
206  static LogicalResult verifyRegionTrait(Operation *op) {
207  // Insist that ops with InnerSymbolTable's provide a Symbol, this is
208  // essential to how InnerRef's work.
209  static_assert(
210  ConcreteType::template hasTrait<::mlir::SymbolOpInterface::Trait>(),
211  "expected operation to define a Symbol");
212 
213  // InnerSymbolTable's must be directly nested within an InnerRefNamespace.
214  auto *parent = op->getParentOp();
215  if (!parent || !parent->hasTrait<InnerRefNamespace>())
216  return op->emitError(
217  "InnerSymbolTable must have InnerRefNamespace parent");
218 
219  return success();
220  }
221 };
