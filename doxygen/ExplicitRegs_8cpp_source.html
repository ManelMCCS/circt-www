27 private:
28  // Recursively routes value v backwards through the pipeline, adding new
29  // registers to 'stage' if the value was not already registered in the stage.
30  // Returns the registerred version of 'v' through 'stage'.
31  Value routeThroughStage(Value v, Block *stage);

33  // Returns the distance between two stages in the pipeline. The distance is
34  // defined wrt. the ordered stages of the pipeline.
35  int64_t stageDistance(Block *from, Block *to);

37  struct RoutedValue {
38  Backedge v;
39  // If true, this value is routed through a stage as a register, else
40  // it is routed through a stage as a pass-through.e
41  bool isReg;
42  };
43  // A mapping storing whether a given stage register constains a registerred
44  // version of a given value. The registered version will be a backedge during
45  // pipeline body analysis. Once the entire body has been analyzed, the
46  // pipeline.stage operations will be replaced with pipeline.ss.reg
47  // operations containing the requested regs, and the backedge will be
48  // replaced. MapVector ensures deterministic iteration order, which in turn
49  // ensures determinism during stage op IR emission.
50  DenseMap<Block *, llvm::MapVector<Value, RoutedValue>> stageRegOrPassMap;
51 
52  // A mapping between stages and their index in the pipeline.
53  llvm::DenseMap<Block *, unsigned> stageMap;
54 
55  std::shared_ptr<BackedgeBuilder> bb;
56 };

66 // NOLINTNEXTLINE(misc-no-recursion)
67 Value ExplicitRegsPass::routeThroughStage(Value v, Block *stage) {
68  Value retVal = v;
69  Block *definingStage = retVal.getParentBlock();
70 
71  // Is the value defined in the current stage?
72  if (definingStage == stage)
73  return retVal;
74 
75  auto regIt = stageRegOrPassMap[stage].find(retVal);
76  if (regIt != stageRegOrPassMap[stage].end()) {
77  // 'v' is already routed through 'stage' - return the registered/passed
78  // version.
79  return regIt->second.v;
80  }
81 
82  // Is the value a constant? If so, we allow it; constants are special cases
83  // which are allowed to be used in any stage.
84  auto *definingOp = retVal.getDefiningOp();
85  if (definingOp && definingOp->hasTrait<OpTrait::ConstantLike>())
86  return retVal;
87 
88  // Value is defined somewhere before the provided stage - route it through the
89  // stage, and recurse to the predecessor stage.
90  int64_t valueLatency = 0;
91  if (auto latencyOp = dyn_cast_or_null<LatencyOp>(definingOp))
92  valueLatency = latencyOp.getLatency();
93 
94  // A value should be registered in this stage if the latency of the value
95  // is less than the distance between the current stage and the defining stage.
96  bool isReg = valueLatency < stageDistance(definingStage, stage);
97  auto valueBackedge = bb->get(retVal.getType());
98  stageRegOrPassMap[stage].insert({retVal, {valueBackedge, isReg}});
99  retVal = valueBackedge;
100 
101  // Recurse - recursion will only create a new backedge if necessary.
102  Block *stagePred = stage->getSinglePredecessor();
103  assert(stagePred && "Expected stage to have a single predecessor");
104  routeThroughStage(v, stagePred);
105  return retVal;
106 }

108 void ExplicitRegsPass::runOnOperation() {
109  ScheduledPipelineOp pipeline = getOperation();
110  OpBuilder b(getOperation().getContext());
111  bb = std::make_shared<BackedgeBuilder>(b, getOperation().getLoc());
112 
113  // Cache external-like inputs in a set for fast lookup. This also includes
114  // clock, reset, and stall.
115  llvm::DenseSet<Value> extLikeInputs;
116  for (auto extInput : pipeline.getInnerExtInputs())
117  extLikeInputs.insert(extInput);
118  extLikeInputs.insert(pipeline.getInnerClock());
119  extLikeInputs.insert(pipeline.getInnerReset());
120  if (pipeline.hasStall())
121  extLikeInputs.insert(pipeline.getInnerStall());
122 
123  // Iterate over the pipeline body in-order (!).
124  stageMap = pipeline.getStageMap();
125  for (Block *stage : pipeline.getOrderedStages()) {
126  // Walk the stage body - we do this since register materialization needs
127  // to consider all levels of nesting within the stage.
128  stage->walk([&](Operation *op) {
129  // Check the operands of this operation to see if any of them cross a
130  // stage boundary.
131  for (OpOperand &operand : op->getOpOperands()) {
132  if (extLikeInputs.contains(operand.get())) {
133  // Never route external inputs through a stage.
134  continue;
135  }
136  if (getParentStageInPipeline(pipeline, operand.get()) == stage) {
137  // The operand is defined by some operation or block which ultimately
138  // resides within the current pipeline stage. No routing needed.
139  continue;
140  }
141  Value reroutedValue = routeThroughStage(operand.get(), stage);
142  if (reroutedValue != operand.get())
143  op->setOperand(operand.getOperandNumber(), reroutedValue);
144  }
145  });
146  }
147 
148  auto *ctx = &getContext();
149 
150  // All values have been recorded through the stages. Now, add registers to the
151  // stage blocks.
152  for (auto &[stage, regMap] : stageRegOrPassMap) {
153  // Gather register inputs to this stage, either from a predecessor stage
154  // or from the original op.
155  llvm::SmallVector<Value> regIns, passIns;
156  Block *predecessorStage = stage->getSinglePredecessor();
157  auto predStageRegOrPassMap = stageRegOrPassMap.find(predecessorStage);
158  assert(predecessorStage && "Stage should always have a single predecessor");
159  for (auto &[value, backedge] : regMap) {
160  if (predStageRegOrPassMap != stageRegOrPassMap.end()) {
161  // Grab the value if passed through the predecessor stage, else,
162  // use the raw value.
163  auto predRegIt = predStageRegOrPassMap->second.find(value);
164  if (predRegIt != predStageRegOrPassMap->second.end()) {
165  if (backedge.isReg)
166  regIns.push_back(predRegIt->second.v);
167  else
168  passIns.push_back(predRegIt->second.v);
169  continue;
170  }
171  }
172 
173  // Not passed through the stage - must be the original value.
174  if (backedge.isReg)
175  regIns.push_back(value);
176  else
177  passIns.push_back(value);
178  }
179 
180  // Replace the predecessor stage terminator, which feeds this stage, with
181  // a new terminator that has materialized arguments.
182  StageOp terminator = cast<StageOp>(predecessorStage->getTerminator());
183  b.setInsertionPoint(terminator);
184  b.create<StageOp>(terminator.getLoc(), terminator.getNextStage(), regIns,
185  passIns);
186  terminator.erase();
187 
188  // ... add arguments to the next stage. Registers first, then passthroughs.
189  llvm::SmallVector<Type> regAndPassTypes;
190  llvm::append_range(regAndPassTypes, ValueRange(regIns).getTypes());
191  llvm::append_range(regAndPassTypes, ValueRange(passIns).getTypes());
192  for (auto [i, type] : llvm::enumerate(regAndPassTypes))
193  stage->insertArgument(i, type, UnknownLoc::get(ctx));
194 
195  // Replace backedges for the next stage with the new arguments.
196  for (auto it : llvm::enumerate(regMap)) {
197  auto index = it.index();
198  auto &[value, backedge] = it.value();
199  backedge.v.setValue(stage->getArgument(index));
200  }
201  }
202 
203  // Clear internal state. See https://github.com/llvm/circt/issues/3235
204  stageRegOrPassMap.clear();
205 }
