273 /// Trait for types which have a width.
274 /// Users must implement:
275 /// ```c++
276 /// /// Return the width if known, or -1 if unknown.
277 /// int32_t getWidthOrSentinel();
278 /// ```
279 template <typename ConcreteType>
280 class WidthQualifiedTypeTrait
281  : public mlir::TypeTrait::TraitBase<ConcreteType, WidthQualifiedTypeTrait> {
282 public:
283  /// Return an optional containing the width, if the width is known (or empty
284  /// if width is unknown).
285  std::optional<int32_t> getWidth() {
286  auto width = static_cast<ConcreteType *>(this)->getWidthOrSentinel();
287  if (width < 0)
288  return std::nullopt;
289  return width;
290  }
291 
292  /// Return true if this integer type has a known width.
293  bool hasWidth() {
294  return 0 <= static_cast<ConcreteType *>(this)->getWidthOrSentinel();
295  }
296 };

400 /// A struct to check if there is a type derived from FIRRTLBaseType.
401 /// `ContainAliasableTypes<BaseTy>::value` returns true if `BaseTy` is derived
402 /// from `FIRRTLBaseType` and not `FIRRTLBaseType` itself, or is not FIRRTL type
403 /// to cover type interfaces.
404 template <typename head, typename... tail>
405 class ContainAliasableTypes {
406 public:
407  static constexpr bool value = ContainAliasableTypes<head>::value ||
408  ContainAliasableTypes<tail...>::value;
409 };
410 
411 template <typename BaseTy>
412 class ContainAliasableTypes<BaseTy> {
413  static constexpr bool isFIRRTLBaseType =
414  std::is_base_of<FIRRTLBaseType, BaseTy>::value &&
415  !std::is_same_v<FIRRTLBaseType, BaseTy>;
416  static constexpr bool isFIRRTLType =
417  std::is_base_of<FIRRTLType, BaseTy>::value;
418 
419 public:
420  static constexpr bool value = isFIRRTLBaseType || !isFIRRTLType;
421 };

423 template <typename... BaseTy>
424 bool type_isa(Type type) { // NOLINT(readability-identifier-naming)
425  // First check if the type is the requested type.
426  if (isa<BaseTy...>(type))
427  return true;
428 
429  // If the requested type is a subtype of FIRRTLBaseType, then check if it is a
430  // type alias wrapping the requested type.
431  if constexpr (ContainAliasableTypes<BaseTy...>::value) {
432  if (auto alias = dyn_cast<BaseTypeAliasType>(type))
433  return type_isa<BaseTy...>(alias.getInnerType());
434  }
435 
436  return false;
437 }
438 
439 // type_isa for a nullable argument.
440 template <typename... BaseTy>
441 bool type_isa_and_nonnull(Type type) { // NOLINT(readability-identifier-naming)
442  if (!type)
443  return false;
444  return type_isa<BaseTy...>(type);
445 }
446 
447 template <typename BaseTy>
448 BaseTy type_cast(Type type) { // NOLINT(readability-identifier-naming)
449  assert(type_isa<BaseTy>(type) && "type must convert to requested type");
450 
451  // If the type is the requested type, return it.
452  if (isa<BaseTy>(type))
453  return cast<BaseTy>(type);
454 
455  // Otherwise, it must be a type alias wrapping the requested type.
456  if constexpr (ContainAliasableTypes<BaseTy>::value) {
457  if (auto alias = dyn_cast<BaseTypeAliasType>(type))
458  return type_cast<BaseTy>(alias.getInnerType());
459  }
460 
461  // Otherwise, it should fail. `cast` should cause a better assertion failure,
462  // so just use it.
463  return cast<BaseTy>(type);
464 }
465 
466 template <typename BaseTy>
467 BaseTy type_dyn_cast(Type type) { // NOLINT(readability-identifier-naming)
468  if (type_isa<BaseTy>(type))
469  return type_cast<BaseTy>(type);
470  return {};
471 }
472 
473 template <typename BaseTy>
474 BaseTy
475 type_dyn_cast_or_null(Type type) { // NOLINT(readability-identifier-naming)
476  if (type_isa_and_nonnull<BaseTy>(type))
477  return type_cast<BaseTy>(type);
478  return {};
479 }

485 /// This class implements the same functionality as TypeSwitch except that
486 /// it uses firrtl::type_dyn_cast for dynamic cast. llvm::TypeSwitch is not
487 /// customizable so this class currently duplicates the code.
488 template <typename T, typename ResultT = void>
489 class FIRRTLTypeSwitch
490  : public llvm::detail::TypeSwitchBase<FIRRTLTypeSwitch<T, ResultT>, T> {
491 public:
492  using BaseT = llvm::detail::TypeSwitchBase<FIRRTLTypeSwitch<T, ResultT>, T>;
493  using BaseT::BaseT;
494  using BaseT::Case;
495  FIRRTLTypeSwitch(FIRRTLTypeSwitch &&other) = default;
496 
497  /// Add a case on the given type.
498  template <typename CaseT, typename CallableT>
499  FIRRTLTypeSwitch<T, ResultT> &
500  Case(CallableT &&caseFn) { // NOLINT(readability-identifier-naming)
501  if (result)
502  return *this;
503 
504  // Check to see if CaseT applies to 'value'. Use `type_dyn_cast` here.
505  if (auto caseValue = circt::firrtl::type_dyn_cast<CaseT>(this->value))
506  result.emplace(caseFn(caseValue));
507  return *this;
508  }
509 
510  /// As a default, invoke the given callable within the root value.
511  template <typename CallableT>
512  [[nodiscard]] ResultT
513  Default(CallableT &&defaultFn) { // NOLINT(readability-identifier-naming)
514  if (result)
515  return std::move(*result);
516  return defaultFn(this->value);
517  }
518 
519  /// As a default, return the given value.
520  [[nodiscard]] ResultT
521  Default(ResultT defaultResult) { // NOLINT(readability-identifier-naming)
522  if (result)
523  return std::move(*result);
524  return defaultResult;
525  }
526 
527  [[nodiscard]] operator ResultT() {
528  assert(result && "Fell off the end of a type-switch");
529  return std::move(*result);
530  }
531 
532 private:
533  /// The pointer to the result of this switch statement, once known,
534  /// null before that.
535  std::optional<ResultT> result;
536 };

538 /// Specialization of FIRRTLTypeSwitch for void returning callables.
539 template <typename T>
540 class FIRRTLTypeSwitch<T, void>
541  : public llvm::detail::TypeSwitchBase<FIRRTLTypeSwitch<T, void>, T> {
542 public:
543  using BaseT = llvm::detail::TypeSwitchBase<FIRRTLTypeSwitch<T, void>, T>;
544  using BaseT::BaseT;
545  using BaseT::Case;
546  FIRRTLTypeSwitch(FIRRTLTypeSwitch &&other) = default;
547 
548  /// Add a case on the given type.
549  template <typename CaseT, typename CallableT>
550  FIRRTLTypeSwitch<T, void> &
551  Case(CallableT &&caseFn) { // NOLINT(readability-identifier-naming)
552  if (foundMatch)
553  return *this;
554 
555  // Check to see if any of the types apply to 'value'.
556  if (auto caseValue = circt::firrtl::type_dyn_cast<CaseT>(this->value)) {
557  caseFn(caseValue);
558  foundMatch = true;
559  }
560  return *this;
561  }
562 
563  /// As a default, invoke the given callable within the root value.
564  template <typename CallableT>
565  void Default(CallableT &&defaultFn) { // NOLINT(readability-identifier-naming)
566  if (!foundMatch)
567  defaultFn(this->value);
568  }
569 
570 private:
571  /// A flag detailing if we have already found a match.
572  bool foundMatch = false;
573 };
