34 // Inlines the module pointed to by 'inst' if the module is empty. This assumes
35 // that 'inst' is the only user of the module. Furthermore, will remove the
36 // inlined module(!). Should probably implement some more generic inlining code
37 // for this, but it's simple enough to do when we know that the module is empty.
38 static void inlineAndEraseIfEmpty(hw::InstanceOp inst) {
39  auto mod = cast<hw::HWModuleLike>(inst.getReferencedModule());
40  if (mod->getNumRegions() == 0)
41  return; // Nothing to do.
42 
43  Block &body = mod->getRegion(0).front();
44  auto &ops = body.getOperations();
45  if (ops.size() > 1)
46  return; // non-empty
47 
48  hw::OutputOp output = cast<hw::OutputOp>(ops.front());
49  DenseMap<Value, Value> valueMapping;
50  for (auto [instOperand, modOperand] :
51  llvm::zip(inst.getOperands(), output.getOperands()))
52  valueMapping[modOperand] = instOperand;
53  llvm::SmallVector<Value> returnValues;
54  for (auto result : output.getOperands())
55  returnValues.push_back(valueMapping[result]);
56 
57  // Replace the instance results with the mapped return values.
58  inst.replaceAllUsesWith(returnValues);
59  inst.erase();
60  mod.erase();
61 }

79  // Arguments used for emitting the body of a stage module. These values must

83  Value enable;

89  // Arguments used for returning the results from a stage. These values must
90  // be within the scope of the stage module body.

93  llvm::SmallVector<Value> passthroughs;
94  Value valid;
95  };

98  lowerStage(Block *stage, StageArgs args, size_t stageIndex,

101  StageReturns emitStageBody(Block *stage, StageArgs args,
102  llvm::ArrayRef<Attribute> registerNames,
103  size_t stageIndex = -1) {
104  assert(args.enable && "enable not set");
105  auto *terminator = stage->getTerminator();
106 
107  // Move the stage operations into the current insertion point.
108  for (auto &op : llvm::make_early_inc_range(*stage)) {
109  if (&op == terminator)
110  continue;
111 
112  if (auto latencyOp = dyn_cast<LatencyOp>(op)) {
113  // For now, just directly emit the body of the latency op. The latency
114  // op is mainly used during register materialization. At a later stage,
115  // we may want to add some TCL-related things here to communicate
116  // multicycle paths.
117  Block *latencyOpBody = latencyOp.getBodyBlock();
118  for (auto &innerOp :
119  llvm::make_early_inc_range(latencyOpBody->without_terminator()))
120  innerOp.moveBefore(builder.getInsertionBlock(),
121  builder.getInsertionPoint());
122  latencyOp.replaceAllUsesWith(
123  latencyOpBody->getTerminator()->getOperands());
124  latencyOp.erase();
125  } else {
126  op.moveBefore(builder.getInsertionBlock(), builder.getInsertionPoint());
127  }
128  }
129 
130  StageReturns rets;
131  auto stageOp = dyn_cast<StageOp>(terminator);
132  if (!stageOp) {
133  assert(isa<ReturnOp>(terminator) && "expected ReturnOp");
134  // This was the pipeline return op - the return op/last stage doesn't
135  // register its operands, hence, all return operands are passthrough
136  // and the valid signal is equal to the unregistered enable signal.
137  rets.passthroughs = terminator->getOperands();
138  rets.valid = args.enable;
139  return rets;
140  }
141 
142  assert(registerNames.size() == stageOp.getRegisters().size() &&
143  "register names and registers must be the same size");
144 
145  // Build data registers.
146  auto stageRegPrefix = getStageRegPrefix(stageIndex);
147  auto loc = stageOp->getLoc();
148 
149  // Build the clock enable signal: enable && !stall (if applicable)
150  Value stageValidAndNotStalled = args.enable;
151  Value notStalled;
152  bool hasStall = static_cast<bool>(args.stall);
153  if (hasStall) {
154  notStalled = comb::createOrFoldNot(loc, args.stall, builder);
155  stageValidAndNotStalled =
156  builder.create<comb::AndOp>(loc, stageValidAndNotStalled, notStalled);
157  }
158 
159  Value notStalledClockGate;
160  if (this->clockGateRegs) {
161  // Create the top-level clock gate.
162  notStalledClockGate = builder.create<seq::ClockGateOp>(
163  loc, args.clock, stageValidAndNotStalled, /*test_enable=*/Value());
164  }
165 
166  for (auto it : llvm::enumerate(stageOp.getRegisters())) {
167  auto regIdx = it.index();
168  auto regIn = it.value();
169 
170  StringAttr regName = registerNames[regIdx].cast<StringAttr>();
171  Value dataReg;
172  if (this->clockGateRegs) {
173  // Use the clock gate instead of clock enable.
174  Value currClockGate = notStalledClockGate;
175  for (auto hierClockGateEnable : stageOp.getClockGatesForReg(regIdx)) {
176  // Create clock gates for any hierarchically nested clock gates.
177  currClockGate = builder.create<seq::ClockGateOp>(
178  loc, currClockGate, hierClockGateEnable, /*test_enable=*/Value());
179  }
180  dataReg = builder.create<seq::CompRegOp>(stageOp->getLoc(), regIn,
181  currClockGate, regName);
182  } else {
183  // Only clock-enable the register if the pipeline is stallable.
184  // For non-stallable pipelines, a data register can always be clocked.
185  if (hasStall) {
186  dataReg = builder.create<seq::CompRegClockEnabledOp>(
187  stageOp->getLoc(), regIn, args.clock, stageValidAndNotStalled,
188  regName);
189  } else {
190  dataReg = builder.create<seq::CompRegOp>(stageOp->getLoc(), regIn,
191  args.clock, regName);
192  }
193  }
194  rets.regs.push_back(dataReg);
195  }
196 
197  // Build valid register. The valid register is always reset to 0, and
198  // clock enabled when not stalling.
199  auto validRegName =
200  builder.getStringAttr(stageRegPrefix.strref() + "_valid");
201  Value validRegResetVal =
202  builder.create<hw::ConstantOp>(terminator->getLoc(), APInt(1, 0, false))
203  .getResult();
204  if (hasStall) {
205  rets.valid = builder.create<seq::CompRegClockEnabledOp>(
206  loc, builder.getI1Type(), args.enable, args.clock, notStalled,
207  validRegName, args.reset, validRegResetVal, validRegName);
208  } else {
209  rets.valid = builder.create<seq::CompRegOp>(
210  loc, builder.getI1Type(), args.enable, args.clock, validRegName,
211  args.reset, validRegResetVal, validRegName);
212  }
213 
214  rets.passthroughs = stageOp.getPassthroughs();
215  return rets;
216  }

218  // A container carrying all-things stage output naming related.
219  // To avoid overloading 'output's to much (i'm trying to keep that reserved
220  // for "output" ports), this is named "egress".
221  struct StageEgressNames {
222  llvm::SmallVector<Attribute> regNames;
223  llvm::SmallVector<Attribute> outNames;
224  llvm::SmallVector<Attribute> inNames;
225  };

227  // Returns a set of names for the output values of a given stage (registers
228  // and passthrough). If `withPipelinePrefix` is true, the names will be
229  // prefixed with the pipeline name.
230  void getStageEgressNames(size_t stageIndex, Operation *stageTerminator,
231  bool withPipelinePrefix,
232  StageEgressNames &egressNames) {
233  StringAttr pipelineName;
234  if (withPipelinePrefix)
235  pipelineName = getPipelineBaseName();
236 
237  if (auto stageOp = dyn_cast<StageOp>(stageTerminator)) {
238  // Registers...
239  std::string assignedRegName, assignedOutName, assignedInName;
240  for (size_t regi = 0; regi < stageOp.getRegisters().size(); ++regi) {
241  if (auto regName = stageOp.getRegisterName(regi)) {
242  assignedRegName = regName.str();
243  assignedOutName = assignedRegName + "_out";
244  assignedInName = assignedRegName + "_in";
245  } else {
246  assignedRegName =
247  ("stage" + Twine(stageIndex) + "_reg" + Twine(regi)).str();
248  assignedOutName = ("out" + Twine(regi)).str();
249  assignedInName = ("in" + Twine(regi)).str();
250  }
251 
252  if (pipelineName) {
253  assignedRegName = pipelineName.str() + "_" + assignedRegName;
254  assignedOutName = pipelineName.str() + "_" + assignedOutName;
255  assignedInName = pipelineName.str() + "_" + assignedInName;
256  }
257 
258  egressNames.regNames.push_back(builder.getStringAttr(assignedRegName));
259  egressNames.outNames.push_back(builder.getStringAttr(assignedOutName));
260  egressNames.inNames.push_back(builder.getStringAttr(assignedInName));
261  }
262 
263  // Passthroughs
264  for (size_t passi = 0; passi < stageOp.getPassthroughs().size();
265  ++passi) {
266  if (auto passName = stageOp.getPassthroughName(passi)) {
267  assignedOutName = (passName.strref() + "_out").str();
268  assignedInName = (passName.strref() + "_in").str();
269  } else {
270  assignedOutName = ("pass" + Twine(passi)).str();
271  assignedInName = ("pass" + Twine(passi)).str();
272  }
273 
274  if (pipelineName) {
275  assignedOutName = pipelineName.str() + "_" + assignedOutName;
276  assignedInName = pipelineName.str() + "_" + assignedInName;
277  }
278 
279  egressNames.outNames.push_back(builder.getStringAttr(assignedOutName));
280  egressNames.inNames.push_back(builder.getStringAttr(assignedInName));
281  }
282  } else {
283  // For the return op, we just inherit the names of the top-level pipeline
284  // as stage output names.
285  llvm::copy(pipeline.getOutputNames().getAsRange<StringAttr>(),
286  std::back_inserter(egressNames.outNames));
287  }
288  }

293 protected:
294  // Determine a reasonable name for the pipeline. This will affect naming

303  Value parentClk;

305  Value parentRst;
306  // ID of the current pipeline, used for naming.
307  size_t pipelineID;
308  // The current pipeline to be converted.
309  ScheduledPipelineOp pipeline;
310 
311  // The module wherein the pipeline resides.
312  hw::HWModuleOp parentModule;
313 
314  OpBuilder &builder;
315 
316  // If true, will use clock gating for registers instead of input muxing.
317  bool clockGateRegs;
318 
319  // Name of this pipeline - used for naming stages and registers.
320  // Implementation defined.
321  StringAttr pipelineName;

324 class PipelineInlineLowering : public PipelineLowering {
325 public:
326  using PipelineLowering::PipelineLowering;
327 
328  StringAttr getStageRegPrefix(size_t stageIdx) override {
329  return builder.getStringAttr(pipelineName.strref() + "_stage" +
330  Twine(stageIdx));
331  }
332 
333  LogicalResult run() override {
334  pipelineName = getPipelineBaseName();
335 
336  // Replace uses of the pipeline internal inputs with the pipeline inputs.
337  for (auto [outer, inner] :
338  llvm::zip(pipeline.getInputs(), pipeline.getInnerInputs()))
339  inner.replaceAllUsesWith(outer);
340 
341  // Replace uses of the external inputs with the inner external inputs.
342  for (auto [outer, inner] :
343  llvm::zip(pipeline.getExtInputs(), pipeline.getInnerExtInputs()))
344  inner.replaceAllUsesWith(outer);
345 
346  // All operations should go directly before the pipeline op, into the
347  // parent module.
348  builder.setInsertionPoint(pipeline);
349  StageArgs args;
350  args.data = pipeline.getInnerInputs();
351  args.enable = pipeline.getGo();
352  args.clock = pipeline.getClock();
353  args.reset = pipeline.getReset();
354  args.stall = pipeline.getStall();
355  if (failed(lowerStage(pipeline.getEntryStage(), args, 0)))
356  return failure();
357 
358  // Replace uses of clock, reset, and stall.
359  pipeline.getInnerClock().replaceAllUsesWith(pipeline.getClock());
360  pipeline.getInnerReset().replaceAllUsesWith(pipeline.getReset());
361  if (auto stall = pipeline.getStall())
362  pipeline.getInnerStall().replaceAllUsesWith(stall);
363 
364  pipeline.erase();
365  return success();
366  }
367 
368  /// NOLINTNEXTLINE(misc-no-recursion)
369  FailureOr<StageReturns>
370  lowerStage(Block *stage, StageArgs args, size_t stageIndex,
371  llvm::ArrayRef<Attribute> /*inputNames*/ = {}) override {
372  OpBuilder::InsertionGuard guard(builder);
373 
374  if (stage != pipeline.getEntryStage()) {
375  // Replace the internal stage inputs with the provided arguments.
376  for (auto [vInput, vArg] :
377  llvm::zip(pipeline.getStageDataArgs(stage), args.data))
378  vInput.replaceAllUsesWith(vArg);
379  }
380 
381  // Replace the stage valid signal.
382  pipeline.getStageEnableSignal(stage).replaceAllUsesWith(args.enable);
383 
384  // Determine stage egress info.
385  auto nextStage = dyn_cast<StageOp>(stage->getTerminator());
386  StageEgressNames egressNames;
387  if (nextStage)
388  getStageEgressNames(stageIndex, nextStage,
389  /*withPipelinePrefix=*/true, egressNames);
390 
391  // Move stage operations into the current module.
392  builder.setInsertionPoint(pipeline);
393  StageReturns stageRets =
394  emitStageBody(stage, args, egressNames.regNames, stageIndex);
395 
396  if (nextStage) {
397  // Lower the next stage.
398  SmallVector<Value> nextStageArgs;
399  llvm::append_range(nextStageArgs, stageRets.regs);
400  llvm::append_range(nextStageArgs, stageRets.passthroughs);
401  args.enable = stageRets.valid;
402  args.data = nextStageArgs;
403  return lowerStage(nextStage.getNextStage(), args, stageIndex + 1);
404  }
405 
406  // Replace the pipeline results with the return op operands.
407  auto returnOp = cast<pipeline::ReturnOp>(stage->getTerminator());
408  llvm::SmallVector<Value> pipelineReturns;
409  llvm::append_range(pipelineReturns, returnOp.getInputs());
410  // The last stage valid signal is the 'done' output of the pipeline.
411  pipelineReturns.push_back(args.enable);
412  pipeline.replaceAllUsesWith(pipelineReturns);
413  return stageRets;
414  }

425  // Helper class to manage grabbing the various inputs for stage modules.
426  struct PipelineStageMod {
427  PipelineStageMod() = default;
428  PipelineStageMod(PipelineOutlineLowering &parent, Block *stage,
429  hw::HWModuleOp mod, bool withStall,
430  bool isParentPipeline = false) {
431  size_t nStageDataArgs = parent.pipeline.getStageDataArgs(stage).size();
432  inputs = mod.getArguments().take_front(nStageDataArgs);
433  if (isParentPipeline) {
434  // The parent pipeline should always have all external inputs available.
435  extInputs = mod.getArguments().slice(
436  nStageDataArgs, parent.pipeline.getExtInputs().size());
437  } else {
438  llvm::SmallVector<Value> stageExtInputs;
439  parent.getStageExtInputs(stage, stageExtInputs);
440  extInputs =
441  mod.getArguments().slice(nStageDataArgs, stageExtInputs.size());
442  }
443 
444  auto portLookup = mod.getPortLookupInfo();
445  if (withStall)
446  stall = mod.getArgument(*portLookup.getInputPortIndex(kStallPortName));
447  enable = mod.getArgument(*portLookup.getInputPortIndex(kEnablePortName));
448  clock = mod.getArgument(*portLookup.getInputPortIndex(kClockPortName));
449  reset = mod.getArgument(*portLookup.getInputPortIndex(kResetPortName));
450  }
451 
452  ValueRange inputs;
453  ValueRange extInputs;
454  Value enable;
455  Value stall;
456  Value clock;
457  Value reset;
458  };

460  // Iterate over the external inputs to the pipeline, and determine which
461  // stages actually reference them. This will be used to generate the stage
462  // module signatures.
463  void gatherExtInputsToStages() {
464  for (auto extIn : pipeline.getInnerExtInputs()) {
465  for (auto *user : extIn.getUsers())
466  stageExtInputs[user->getBlock()].insert(extIn);
467  }
468  }
469 
470  LogicalResult run() override {
471  pipelineName = StringAttr::get(pipeline.getContext(),
472  parentModule.getName() + "_" +
473  getPipelineBaseName().strref());
474  cloneConstantsToStages();
475 
476  // Map external inputs to names - we use this to generate nicer names for
477  // the stage module arguments.
478  if (!pipeline.getExtInputs().empty()) {
479  for (auto [extIn, extName] :
480  llvm::zip(pipeline.getInnerExtInputs(),
481  pipeline.getExtInputNames()->getAsRange<StringAttr>())) {
482  extInputNames[extIn] = extName;
483  }
484  }
485 
486  // Build the top-level pipeline module.
487  bool withStall = static_cast<bool>(pipeline.getStall());
488  pipelineMod = buildPipelineLike(
489  pipelineName.strref(), pipeline.getInputs().getTypes(),
490  pipeline.getInnerExtInputs(), pipeline.getDataOutputs().getTypes(),
491  withStall, pipeline.getInputNames().getValue(),
492  pipeline.getOutputNames().getValue());
493  auto portLookup = pipelineMod.getPortLookupInfo();
494  pipelineClk = pipelineMod.getBody().front().getArgument(
495  *portLookup.getInputPortIndex(kClockPortName));
496  pipelineRst = pipelineMod.getBody().front().getArgument(
497  *portLookup.getInputPortIndex(kResetPortName));
498 
499  if (withStall)
500  pipelineStall = pipelineMod.getBody().front().getArgument(
501  *portLookup.getInputPortIndex(kStallPortName));
502 
503  if (!pipeline.getExtInputs().empty()) {
504  // Maintain a mapping between external inputs and their corresponding
505  // block argument in the top-level pipeline.
506  auto modInnerExtInputs =
507  pipelineMod.getBody().front().getArguments().slice(
508  pipeline.getExtInputs().getBeginOperandIndex(),
509  pipeline.getExtInputs().size());
510  for (auto [extIn, barg] :
511  llvm::zip(pipeline.getInnerExtInputs(), modInnerExtInputs)) {
512  toplevelExtInputs[extIn] = barg;
513  }
514  }
515 
516  // Instantiate the pipeline in the parent module.
517  builder.setInsertionPoint(pipeline);
518  llvm::SmallVector<Value, 4> pipelineOperands;
519  llvm::append_range(pipelineOperands, pipeline.getInputs());
520  llvm::append_range(pipelineOperands, pipeline.getExtInputs());
521  pipelineOperands.push_back(pipeline.getGo());
522  if (auto stall = pipeline.getStall())
523  pipelineOperands.push_back(stall);
524  llvm::append_range(pipelineOperands,
525  ValueRange{pipeline.getClock(), pipeline.getReset()});
526 
527  auto pipelineInst = builder.create<hw::InstanceOp>(
528  pipeline.getLoc(), pipelineMod,
529  builder.getStringAttr(pipelineMod.getName()), pipelineOperands);
530 
531  // Replace the top-level pipeline results with the pipeline instance
532  // results.
533  pipeline.replaceAllUsesWith(pipelineInst.getResults());
534 
535  // Determine the external inputs to each stage.
536  gatherExtInputsToStages();
537 
538  // From now on, insertion point must point to the pipeline module body.
539  // This ensures that pipeline stage instantiations and free-standing
540  // operations are inserted into the pipeline module.
541  builder.setInsertionPointToStart(pipelineMod.getBodyBlock());
542 
543  pipelineStageMod =
544  PipelineStageMod(*this, pipeline.getEntryStage(), pipelineMod,
545  withStall, /*isParentPipeline*/ true);
546 
547  StageArgs args;
548  args.data = pipelineStageMod.inputs;
549  args.enable = pipelineStageMod.enable;
550  args.clock = pipelineStageMod.clock;
551  args.reset = pipelineStageMod.reset;
552  args.stall = pipelineStageMod.stall;
553  FailureOr<StageReturns> lowerRes = lowerStage(
554  pipeline.getEntryStage(), args, 0, pipeline.getInputNames().getValue());
555  if (failed(lowerRes))
556  return failure();
557 
558  // Assign the output op of the top-level pipeline module.
559  auto outputOp =
560  cast<hw::OutputOp>(pipelineMod.getBodyBlock()->getTerminator());
561  outputOp->setOperands(lowerRes->passthroughs);
562  outputOp->insertOperands(lowerRes->passthroughs.size(), lowerRes->valid);
563  pipeline.erase();
564 
565  // Mini-optimization: the lowerStage/emitBody logic is written to not do any
566  // special-case logic for the final stage of the pipeline.
567  // In many cases, there are no operations in the final stage, except for the
568  // return op. In these cases, we'll just inline the exit stage into the
569  // pipeline module, and erase the (empty) last stage module.
570  inlineAndEraseIfEmpty(currentStageInst);
571 
572  return success();
573  }

575  /// NOLINTNEXTLINE(misc-no-recursion)
576  FailureOr<StageReturns>
577  lowerStage(Block *stage, StageArgs argsToStage, size_t stageIndex,
578  llvm::ArrayRef<Attribute> inputNames = {}) override {
579  hw::OutputOp stageOutputOp;
580  ValueRange nextStageArgs;
581  bool withStall = static_cast<bool>(argsToStage.stall);
582 
583  auto replaceValuesInStage = [&](Value src, Value dst) {
584  src.replaceUsesWithIf(dst, [&](OpOperand &operand) {
585  return operand.getOwner()->getBlock() == stage;
586  });
587  };
588 
589  // Anything but the last stage
590  StageEgressNames egressNames;
591  auto [stageMod, stageInst] =
592  buildStage(stage, argsToStage, stageIndex, inputNames, egressNames);
593  auto thisStageMod = PipelineStageMod{*this, stage, stageMod, withStall};
594  currentStageInst = stageInst;
595 
596  // Remap the internal inputs of the stage to the stage module block
597  // arguments.
598  for (auto [vInput, vBarg] :
599  llvm::zip(pipeline.getStageDataArgs(stage), thisStageMod.inputs))
600  vInput.replaceAllUsesWith(vBarg);
601 
602  // Remap external inputs to the stage to the external inputs in the
603  // module block arguments.
604  llvm::SmallVector<Value> stageExtInputs;
605  getStageExtInputs(stage, stageExtInputs);
606 
607  for (auto [vExtInput, vExtBarg] :
608  llvm::zip(stageExtInputs, thisStageMod.extInputs))
609  replaceValuesInStage(vExtInput, vExtBarg);
610 
611  // Move stage operations into the module.
612  builder.setInsertionPointToStart(&stageMod.getBody().front());
613  stageOutputOp =
614  cast<hw::OutputOp>(stageMod.getBody().front().getTerminator());
615 
616  // As well as any use of clock, reset and stall within this stage.
617  replaceValuesInStage(pipeline.getStageEnableSignal(stage),
618  thisStageMod.enable);
619  replaceValuesInStage(pipeline.getInnerClock(), thisStageMod.clock);
620  replaceValuesInStage(pipeline.getInnerReset(), thisStageMod.reset);
621  if (pipeline.hasStall())
622  replaceValuesInStage(pipeline.getInnerStall(), thisStageMod.stall);
623 
624  // Arguments passed to emitStageBody that map to the inner stage module
625  // arguments.
626  StageArgs innerArgs;
627  innerArgs.data = thisStageMod.inputs;
628  innerArgs.enable = thisStageMod.enable;
629  innerArgs.clock = thisStageMod.clock;
630  innerArgs.reset = thisStageMod.reset;
631  innerArgs.stall = thisStageMod.stall;
632  StageReturns stageRets =
633  emitStageBody(stage, innerArgs, egressNames.regNames, stageIndex);
634 
635  // Assign the output operation to the stage return values.
636  stageOutputOp->insertOperands(0, stageRets.regs);
637  stageOutputOp->insertOperands(stageOutputOp.getNumOperands(),
638  stageRets.passthroughs);
639  stageOutputOp->insertOperands(stageOutputOp.getNumOperands(),
640  stageRets.valid);
641 
642  if (auto stageOp = dyn_cast<StageOp>(stage->getTerminator())) {
643  // Lower the next stage.
644  StageArgs nextStageArgs;
645  nextStageArgs.data = stageInst.getResults().drop_back();
646  nextStageArgs.enable = stageInst.getResults().back();
647  nextStageArgs.clock = pipelineClk;
648  nextStageArgs.reset = pipelineRst;
649  nextStageArgs.stall = pipelineStall;
650  return lowerStage(stageOp.getNextStage(), nextStageArgs, stageIndex + 1,
651  // It is the current stages' egress info which
652  // determines the next stage input names.
653  egressNames.inNames);
654  }
655 
656  // This was the final stage - forward the return values of the last stage
657  // instance as the stage return value.
658  stageRets.passthroughs = stageInst.getResults().drop_back();
659  stageRets.valid = stageInst.getResults().back();
660 
661  return stageRets;
662  }

665  // Creates a clone of constant-like operations within each stage that
666  // references them. This ensures that, once outlined, each stage will
667  // reference valid constants.
668  void cloneConstantsToStages() {
669  // Maintain a mapping of the constants already cloned to a stage.
670  for (auto &constantOp : llvm::make_early_inc_range(pipeline.getOps())) {
671  if (!constantOp.hasTrait<OpTrait::ConstantLike>())
672  continue;
673 
674  llvm::DenseMap<Block *, llvm::SmallVector<OpOperand *>> stageUses;
675  Block *stageWithConstant = constantOp.getBlock();
676  for (auto &use : constantOp.getUses()) {
677  Block *usingStage = use.getOwner()->getBlock();
678  if (usingStage == stageWithConstant)
679  continue;
680  stageUses[usingStage].push_back(&use);
681  }
682 
683  // Clone the constant into each stage that uses it, and replace usages
684  // within that stage.
685  for (auto &[stage, uses] : stageUses) {
686  Operation *clonedConstant = constantOp.clone();
687  builder.setInsertionPointToStart(stage);
688  builder.insert(clonedConstant);
689 
690  clonedConstant->setLoc(constantOp.getLoc());
691  clonedConstant->moveBefore(&stage->front());
692  for (OpOperand *use : uses)
693  use->set(clonedConstant->getResult(0));
694  }
695  }
696  }

698  // Builds a pipeline-like module (top level pipeline and pipeline stages).
699  // The module signature is:
700  // ($ins, $extIns, enable, (stall)? : i1, clk : i1, reset : i1)
701  // -> ($outs, valid : i1)
702  // Optionally, explicit names can be provided for the inputs, outputs, and
703  // external inputs.
704  hw::HWModuleOp buildPipelineLike(Twine name, TypeRange inputs,
705  ValueRange extInputs, TypeRange outputs,
706  bool withStall,
707  llvm::ArrayRef<Attribute> inputNames,
708  llvm::ArrayRef<Attribute> outputNames) {
709  assert(inputs.size() == inputNames.size());
710  assert(outputs.size() == outputNames.size());
711  OpBuilder::InsertionGuard guard(builder);
712  builder.setInsertionPoint(parentModule);
713  llvm::SmallVector<hw::PortInfo> ports;
714 
715  // Data inputs
716  for (auto [idx, in] : llvm::enumerate(inputs))
717  ports.push_back(hw::PortInfo{{inputNames[idx].cast<StringAttr>(), in,
718  hw::ModulePort::Direction::Input}});
719 
720  // External inputs
721  for (auto extIn : extInputs) {
722  ports.push_back(hw::PortInfo{{extInputNames.at(extIn), extIn.getType(),
723  hw::ModulePort::Direction::Input}});
724  }
725 
726  // Enable input
727  ports.push_back(
728  hw::PortInfo{{builder.getStringAttr(kEnablePortName),
729  builder.getI1Type(), hw::ModulePort::Direction::Input}});
730 
731  if (withStall) {
732  // Stall input
733  ports.push_back(hw::PortInfo{{builder.getStringAttr(kStallPortName),
734  builder.getI1Type(),
735  hw::ModulePort::Direction::Input}});
736  }
737 
738  // clock and reset
739  ports.push_back(hw::PortInfo{{
740  builder.getStringAttr(kClockPortName),
741  builder.getI1Type(),
742  hw::ModulePort::Direction::Input,
743  }});
744  ports.push_back(hw::PortInfo{{
745  builder.getStringAttr(kResetPortName),
746  builder.getI1Type(),
747  hw::ModulePort::Direction::Input,
748  }});
749 
750  for (auto [idx, out] : llvm::enumerate(outputs))
751  ports.push_back(hw::PortInfo{{outputNames[idx].cast<StringAttr>(), out,
752  hw::ModulePort::Direction::Output}});
753 
754  // Valid output
755  ports.push_back(
756  hw::PortInfo{{builder.getStringAttr(kValidPortName),
757  builder.getI1Type(), hw::ModulePort::Direction::Output}});
758 
759  return builder.create<hw::HWModuleOp>(pipeline.getLoc(),
760  builder.getStringAttr(name), ports);
761  }

763  std::tuple<hw::HWModuleOp, hw::InstanceOp>
764  buildStage(Block *stage, StageArgs args, size_t stageIndex,
765  llvm::ArrayRef<Attribute> inputNames,
766  StageEgressNames &egressNames) {
767  assert(args.data.size() == inputNames.size());
768  builder.setInsertionPoint(parentModule);
769  llvm::SmallVector<Type> outputTypes;
770  auto *terminator = stage->getTerminator();
771  if (auto stageOp = dyn_cast<StageOp>(terminator)) {
772  // The return values of a stage are the inputs to the next stage.
773  llvm::append_range(outputTypes, ValueRange(pipeline.getStageDataArgs(
774  stageOp.getNextStage()))
775  .getTypes());
776  } else {
777  // The return values of the last stage are the outputs of the pipeline.
778  auto returnOp = cast<ReturnOp>(terminator);
779  llvm::append_range(outputTypes, returnOp.getOperandTypes());
780  }
781  getStageEgressNames(stageIndex, terminator,
782  /*withPipelinePrefix=*/false, egressNames);
783 
784  llvm::SmallVector<Value> stageExtInputs;
785  getStageExtInputs(stage, stageExtInputs);
786  bool hasStall = static_cast<bool>(args.stall);
787  hw::HWModuleOp mod = buildPipelineLike(
788  pipelineName.strref() + "_s" + Twine(stageIndex),
789  ValueRange(pipeline.getStageDataArgs(stage)).getTypes(), stageExtInputs,
790  outputTypes, hasStall, inputNames, egressNames.outNames);
791 
792  // instantiate...
793  OpBuilder::InsertionGuard guard(builder);
794  builder.setInsertionPoint(pipelineMod.getBodyBlock()->getTerminator());
795  llvm::SmallVector<Value, 4> stageOperands;
796  llvm::append_range(stageOperands, args.data);
797 
798  // Gather external inputs for this stage from the top-level pipeline
799  // module.
800  for (auto extInput : stageExtInputs)
801  stageOperands.push_back(toplevelExtInputs[extInput]);
802 
803  stageOperands.push_back(args.enable);
804  if (hasStall)
805  stageOperands.push_back(pipelineStall);
806  stageOperands.push_back(pipelineClk);
807  stageOperands.push_back(pipelineRst);
808  auto inst = builder.create<hw::InstanceOp>(pipeline.getLoc(), mod,
809  mod.getName(), stageOperands);
810 
811  return {mod, inst};
812  }
813 
814  // Pipeline module clock.
815  Value pipelineClk;
816  // Pipeline module reset.
817  Value pipelineRst;
818  // Pipeline module stall.
819  Value pipelineStall;
820 
821  // Handle to the instantiation of the current stage under processing.
822  hw::InstanceOp currentStageInst;
823 
824  // Pipeline module, containing stage instantiations.
825  hw::HWModuleOp pipelineMod;
826 
827  // Handle to the instantiation of the last stage in the pipeline.
828  hw::InstanceOp lastStageInst;
829 
830  // Handle to the PipelineStageMod for the parent pipeline module.
831  PipelineStageMod pipelineStageMod;
832 
833  // A mapping between stages and the external inputs which they reference.
834  // A SetVector is used to ensure determinism in the order of the external
835  // inputs to a stage.
836  llvm::DenseMap<Block *, llvm::SetVector<Value>> stageExtInputs;
837 
838  // A mapping between external inputs and their corresponding name attribute.
839  DenseMap<Value, StringAttr> extInputNames;
840 
841  // A mapping between external inputs and their corresponding top-level
842  // input in the pipeline module.
843  llvm::DenseMap<Value, Value> toplevelExtInputs;
844 
845  // Wrapper around stageExtInputs which returns a llvm::SmallVector<Value>,
846  // which in turn can be used to get a ValueRange (and by extension,
847  // TypeRange).
848  void getStageExtInputs(Block *stage, llvm::SmallVector<Value> &extInputs) {
849  llvm::append_range(extInputs, stageExtInputs[stage]);
850  }

862 void PipelineToHWPass::runOnOperation() {
863  OpBuilder builder(&getContext());
864  // Iterate over each pipeline op in the module and convert.
865  // Note: This pass matches on `hw::ModuleOp`s and not directly on the
866  // `ScheduledPipelineOp` due to the `ScheduledPipelineOp` being erased during
867  // this pass.
868  size_t pipelinesSeen = 0;
869  for (auto pipeline : llvm::make_early_inc_range(
870  getOperation().getOps<ScheduledPipelineOp>())) {
871  if (outlineStages) {
872  if (failed(PipelineOutlineLowering(pipelinesSeen, pipeline, builder,
873  clockGateRegs)
874  .run())) {
875  signalPassFailure();
876  return;
877  }
878  } else if (failed(PipelineInlineLowering(pipelinesSeen, pipeline, builder,
879  clockGateRegs)
880  .run())) {
881  signalPassFailure();
882  return;
883  }
884  ++pipelinesSeen;
885  }
886 }
